def IQR(x):
    """ Calculate IQR Fences"""
    
    Q1 = x.quantile(0.25)
    Q3 = x.quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - (1.5*IQR)
    upper = Q3 + (1.5*IQR)
    upp = max(x[x < upper])
    low = min(x[x > lower])
    return low, upp
